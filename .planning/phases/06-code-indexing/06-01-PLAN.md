  ```typescript
      const parser = new Parser();
      parser.setLanguage(TypeScriptLang.tsx);
      const tree = parser.parse("const App = () => <div>hello</div>;");
      expect(tree.rootNode.type).toBe("program");
    });

    test("Python parser parses a function", () => {
      const parser = new Parser();
      parser.setLanguage(PythonLang);
      const tree = parser.parse("def hello():\n    return 1\n");
      expect(tree.rootNode.type).toBe("module");

    test("Rust parser parses a function", () => {
      const parser = new Parser();
      parser.setLanguage(RustLang);
      const tree = parser.parse("fn hello() -> i32 { 1 }");
      expect(tree.rootNode.type).toBe("source_file");
    });
  });
  ```

     ```typescript
     ];
     ```

  2. `SUPPORTED_EXTENSIONS` — the file extensions to scan:
     ```typescript
     export const SUPPORTED_EXTENSIONS = [".ts", ".tsx", ".py", ".rs"] as const;
     ```

  3. `isTestFile(relPath: string): boolean` — detect test files per user decision:
     - Match patterns: `*.test.*`, `*.spec.*`, `test_*`, `*_test.*`
     - Use path.basename() for checking; handle full relative paths
     - Return true if the file matches any test pattern

  4. `scanFiles(projectRoot: string, opts?: ScanOptions): Promise<ScanResult>`:
     ```typescript
     interface ScanOptions {
       exclude_patterns?: string[];
       include_patterns?: string[];
     }

     interface ScanResult {
       files: FileEntry[];
       files_scanned: number;
     }

     interface FileEntry {
       relativePath: string;  // always forward-slash separated, relative to projectRoot
       absolutePath: string;
       language: "typescript" | "python" | "rust";
       isTest: boolean;
     }
     ```

     Implementation:
     1. Build an `ignore` instance from:
        - `.gitignore` in projectRoot (if exists)
        - `DEFAULT_EXCLUSIONS`
        - `opts.exclude_patterns` (if provided)
     2. Use `Bun.Glob` to scan `**/*.{ts,tsx,py,rs}` from projectRoot
     3. For each file, check `ig.ignores(relPath)` — skip if ignored
     4. If `include_patterns` provided, also check that the file matches at least one include pattern (use Bun.Glob or regex)
     5. Determine language from extension: `.ts`/`.tsx` → "typescript", `.py` → "python", `.rs` → "rust"
     6. Check `isTestFile(relPath)` and tag result
     7. Ensure all paths use forward slashes (normalize with `relPath.replace(/\\/g, '/')`)
     8. Log per-file progress to stderr: `logger.debug({ file: relPath }, "Scanning file")`

  **parser.ts — Parser initialization and dispatch:**

  Exports:
  1. `getParserForFile(filePath: string): Parser` — returns a configured parser:

     - `.py` → Parser with PythonLang
     - `.rs` → Parser with RustLang
     - Other → throw Error("Unsupported file extension: {ext}")
     - Create parser instances lazily (module-level cache, one per language variant)

  2. `parseSource(filePath: string, source: string): Parser.Tree` — parse source code:
     - Get parser via `getParserForFile(filePath)`
     - Call `parser.parse(source)`
     - Return the tree
     - If tree is null (should not happen), throw Error
