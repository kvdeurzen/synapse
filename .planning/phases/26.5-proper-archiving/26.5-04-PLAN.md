    **1d. Create packages/framework/config/cliff.toml:**
    A git-cliff configuration template that maps conventional commit types to changelog sections:
    ```toml
    # git-cliff configuration template for Synapse-managed projects
    # Install: `cargo install git-cliff` or `brew install git-cliff`
    # Usage: `git cliff -o CHANGELOG.md`
    # Note: The Document Controller generates changelogs independently of git-cliff.
    # This file is a convenience for users who want standalone changelog generation.

    [changelog]
    header = "# Changelog\n\nAll notable changes to this project will be documented in this file.\n"
    body = """
    {% if version %}\
        ## [{{ version | trim_start_matches(pat="v") }}] - {{ timestamp | date(format="%Y-%m-%d") }}
    {% else %}\
        ## [unreleased]
    {% endif %}\
    {% for group, commits in commits | group_by(attribute="group") %}
        ### {{ group | striptags | trim | upper_first }}
        {% for commit in commits %}
            - {{ commit.message | upper_first }}\
        {% endfor %}
    {% endfor %}\n
    """
    trim = true

    [git]
    conventional_commits = true
    filter_unconventional = true
    commit_parsers = [
      { message = "^feat", group = "Features" },
      { message = "^fix", group = "Bug Fixes" },
      { message = "^refactor", group = "Refactoring" },
      { message = "^docs", group = "Documentation" },
      { message = "^test", group = "Testing" },
      { message = "^perf", group = "Performance" },
      { message = "^style", group = "Styling" },
      { message = "^chore", group = "Miscellaneous" },
      { message = "^ci", group = "CI/CD" },
      { message = "^build", group = "Build" },
    ]
    filter_commits = false
    tag_pattern = "v[0-9].*"
    sort_commits = "oldest"
    ```
