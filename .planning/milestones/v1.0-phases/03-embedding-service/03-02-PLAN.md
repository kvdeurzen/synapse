**Health check function:**
```typescript
export async function checkOllamaHealth(
  ollamaUrl: string,
  embedModel: string,
): Promise<OllamaHealthStatus> {
  try {
    const response = await fetch(`${ollamaUrl}/api/tags`, {
      signal: AbortSignal.timeout(5_000), // 5-second timeout per user decision
    });
    if (!response.ok) return 'unreachable';
    const data = await response.json() as { models: Array<{ name: string }> };
    // Model name may include tag suffix (e.g., "nomic-embed-text:latest") — Pitfall 2
    const hasModel = data.models.some(
      (m) => m.name === embedModel || m.name.startsWith(`${embedModel}:`)
    );
    return hasModel ? 'ok' : 'model_missing';
  } catch {
    return 'unreachable';
  }
}
```

Per user decision: "Checks Ollama reachability AND verifies nomic-embed-text model is available (no test embed)". Uses GET /api/tags which is metadata-only and fast (Pitfall 3: avoid test embed on startup).

**Update `src/server.ts`:**

Change `startServer` signature to accept `config`:

```typescript
export async function startServer(server: McpServer, config: SynapseConfig): Promise<void> {
  // Blocking health check — server waits for result before connecting transport
  // Per user decision: "Blocking health check with 5-second timeout — server waits for result before registering tools"
  const status = await checkOllamaHealth(config.ollamaUrl, config.embedModel);
  setOllamaStatus(status);

  if (status !== 'ok') {
    logger.warn(
      { ollamaUrl: config.ollamaUrl, embedModel: config.embedModel, status },
      status === 'unreachable'
        ? `Ollama unreachable at ${config.ollamaUrl}. Run: ollama serve`
        : `Model ${config.embedModel} not found. Run: ollama pull ${config.embedModel}`,
    );
  } else {
    logger.info({ ollamaUrl: config.ollamaUrl, embedModel: config.embedModel }, 'Ollama health check passed');
  }

  const transport = new StdioServerTransport();
  await server.connect(transport);
  logger.info("Synapse MCP server running on stdio");
}
```

Import `checkOllamaHealth`, `setOllamaStatus` from `./services/embedder.js`. The key behavior: if health check returns `unreachable` or `model_missing`, log a warning but DO NOT throw — the server must still start (user decision: "server starts anyway").
