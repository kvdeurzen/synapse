```typescript
import * as lancedb from "@lancedb/lancedb";
import { mkdirSync } from "node:fs";
import { resolve } from "node:path";
import { logger } from "../logger.js";

export async function connectDb(dbPath: string): Promise<lancedb.Connection> {
  const absPath = resolve(dbPath);
  mkdirSync(absPath, { recursive: true });
  const db = await lancedb.connect(absPath);
  logger.debug({ dbPath: absPath }, "Connected to LanceDB");
  return db;
}
```

Note: connectDb() already handles resolve(), mkdirSync(), and lancedb.connect() in one call. The callers should pass the raw dbPath (relative or absolute) and connectDb() returns the connection handle. The callers do NOT need to resolve the path themselves since connectDb() does it internally. However, initProject currently stores the resolved path in its return value (`database_path: absPath`). After switching to connectDb(), the caller needs the resolved path for the return value. Use `resolve(dbPath)` from node:path to compute the return value separately, or extract it before calling connectDb().

**Gap being closed:** The BTree index test (line 90-99 of test/db/init-project.test.ts) asserts `Array.isArray(indices)` which is always true regardless of whether indexes exist. This proves nothing about BTree index creation.

**Current test code (lines 90-99):**
```typescript
test("creates BTree index on project_id for each table", async () => {
  await initProject(tmpDir, "proj");
  const db = await lancedb.connect(tmpDir);
  for (const name of TABLE_NAMES) {
    const table = await db.openTable(name);
    const indices = await table.listIndices();
    // BTree index on project_id should exist (or the attempt was gracefully handled)
    // We verify at minimum that listIndices() is callable and returns an array
    expect(Array.isArray(indices)).toBe(true);
  }
});
```

**Replace with strengthened assertion:**
```typescript
test("creates BTree index on project_id for each table", async () => {
  await initProject(tmpDir, "proj");
  const db = await lancedb.connect(tmpDir);
  for (const name of TABLE_NAMES) {
    const table = await db.openTable(name);
    const indices = await table.listIndices();
    // BTree index on project_id should exist — verify the array is non-empty
    // If the index was created successfully, listIndices() returns at least one entry
    // If it gracefully failed (Pitfall 3), this test documents the actual behavior
    expect(indices.length).toBeGreaterThan(0);
  }
});
```

**Why `indices.length > 0` instead of checking for a specific "project_id" index name:**
- LanceDB's `listIndices()` return type varies across versions. In 0.26.2 it returns an array of index metadata objects, but the exact field names for column reference are not guaranteed in the public API docs.
- Asserting `length > 0` is the right balance: it fails if no indexes were created (which would mean the try/catch swallowed a real failure silently), but it does not over-specify the index metadata format.
- Since initProject only creates one index per table (on project_id), a non-empty array confirms the index exists.
