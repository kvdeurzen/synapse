Implementation steps:
1. Open DB via connectDb(dbPath).
2. Query project_meta table for `last_index_at`:
   ```typescript
   const metaTable = await db.openTable("project_meta");
   const metaRows = await metaTable.query()
     .where(`project_id = '${projectId}'`).limit(1).toArray();
   const lastIndexAt = metaRows.length > 0 ? (metaRows[0].last_index_at as string | null) : null;
   ```
3. Query all code_chunks for this project:
   ```typescript
   const codeChunksTable = await db.openTable("code_chunks");
   const allRows = await codeChunksTable.query()
     .where(`project_id = '${projectId}'`)
     .select(["file_path", "language", "file_hash", "chunk_id"])
     .toArray();
   ```
4. Build per-language breakdown:
   ```typescript
   const langMap = new Map<string, { files: Set<string>; chunks: number }>();
   for (const row of allRows) {
     const lang = (row.language as string) ?? "unknown";
     if (!langMap.has(lang)) langMap.set(lang, { files: new Set(), chunks: 0 });
     const entry = langMap.get(lang)!;
     entry.files.add(row.file_path as string);
     entry.chunks++;
   }
   ```
5. Calculate total_files (distinct file_paths) and total_chunks:
   ```typescript
   const allFiles = new Set(allRows.map(r => r.file_path as string));
   const totalFiles = allFiles.size;
   const totalChunks = allRows.length;
   ```
6. Staleness check (only if projectRoot provided):
   ```typescript
   import { createHash } from "node:crypto";
   let staleFiles: number | null = null;
   if (projectRoot) {
     staleFiles = 0;
     const fileHashMap = new Map<string, string>(); // file_path -> file_hash
     for (const row of allRows) {
       if (!fileHashMap.has(row.file_path as string)) {
         fileHashMap.set(row.file_path as string, row.file_hash as string);
       }
     }
     for (const [filePath, storedHash] of fileHashMap) {
       try {
         const absPath = join(projectRoot, filePath);
         const content = await Bun.file(absPath).text();
         const currentHash = createHash("sha256").update(content).digest("hex");
         if (currentHash !== storedHash) staleFiles++;
       } catch {
         staleFiles++; // file deleted = stale
       }
     }
   }
   ```
   Import `join` from `node:path`.
   CRITICAL: When project_root is NOT provided, return `stale_files: null` (NOT 0 — that would be misleading per CONTEXT.md and Pitfall 6).

7. Build and return result (facts-only, no recommendations per CONTEXT.md).

       const codeTable = await db.openTable("code_chunks");

         .limit(100)

       totalCodeItems = codeRows.length;

         const content = row.content as string;
         const summary = extractSnippet(content, "", 100); // signature + first comment
         const summaryTokens = countTokens(summary);

         if (totalTokens + summaryTokens > maxTokens) {
