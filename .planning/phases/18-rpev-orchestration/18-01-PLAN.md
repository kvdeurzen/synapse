Update `packages/framework/hooks/synapse-startup.js` to read the `[rpev.involvement]` matrix and `[rpev.domain_overrides]` from trust.toml and inject them into `additionalContext`. This follows the identical pattern used for `tierContext` injection.

**Where to add the code:** After the `tierContext` block (after the `catch (configErr)` block around line 180), and before the final `contextParts` assembly (around line 183).

**What to add:**

1. Inside the existing `if (trustToml && agentsToml)` block (where tierContext is built), add a new `rpevContext` variable built from the trust.toml data:

```javascript
// --- RPEV involvement matrix injection ---
let rpevContext = "";
if (trustToml && trustToml.rpev) {
  const involvement = trustToml.rpev.involvement || {};
  const domainOverrides = trustToml.rpev.domain_overrides || {};
  const gateLevels = trustToml.rpev.explicit_gate_levels || ["project", "epic"];

  const rpevLines = [
    "",
    "## RPEV Involvement Matrix (from trust.toml)",
    "",
    "Your behavior at each RPEV stage is governed by this matrix.",
    "Modes:",
    "  drives    = user initiates the action",
    "  co-pilot  = agent proposes, user approves",
    "  reviews   = agent does, user reviews output",
    "  autopilot = agent does, no user involvement",
    "  monitors  = agent does, user notified + can intervene",
    "",
  ];

  // Group by level for readable display
  const levels = ["project", "epic", "feature", "work_package"];
  const stages = ["refine", "plan", "execute", "validate"];
  for (const level of levels) {
    const entries = stages
      .map((stage) => {
        const key = `${level}_${stage}`;
        return involvement[key] ? `${stage}=${involvement[key]}` : null;
      })
      .filter(Boolean);
    if (entries.length > 0) {
      rpevLines.push(`  ${level}: ${entries.join(", ")}`);
    }
  }

  rpevLines.push(
    "",
    `Explicit gate levels (user must signal readiness): ${gateLevels.join(", ")}`,
  );

  if (Object.keys(domainOverrides).length > 0) {
    rpevLines.push("", "Domain overrides (escalate involvement for specific domains):");
    for (const [key, mode] of Object.entries(domainOverrides)) {
      rpevLines.push(`  ${key}: ${mode}`);
    }
  }

  rpevContext = rpevLines.join("\n");
}
```

2. Declare `rpevContext` variable at the same scope level as `tierContext` (outside the try/catch). Initialize it as `let rpevContext = "";` near where `tierContext` is declared.

```javascript
if (rpevContext) { contextParts.push(rpevContext); }
const additionalContext = contextParts.join("\n\n");
```

IMPORTANT: The rpevContext block must be INSIDE the same try/catch as the tierContext block (it uses `trustToml` which is already in scope there). Do not create a separate try/catch.

Do NOT change existing baseInstructions, projectContext, or tierContext logic — only add the rpevContext block.
